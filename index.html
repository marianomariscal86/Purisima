// ===================================================================== // Edge Function: tiie-fetch // Consulta la TIIE 28 días vigente directo de Banxico (API SIE) y la // guarda en la tabla tasa_config. El token de Banxico se guarda como // "secret" del proyecto (BANXICO_TOKEN) — nunca queda expuesto en el // código del front-end ni en GitHub. // Solo usuarios con rol 'admin' pueden ejecutarla (se valida su JWT). // ===================================================================== import { createClient } from "jsr:@supabase/supabase-js@2"; const SERIE_TIIE_28 = "SF60648"; Deno.serve(async (req) => { const cors = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", }; if (req.method === "OPTIONS") return new Response("ok", { headers: cors }); try { // ---- Verifica que quien llama esté autenticado y sea admin ---- const authHeader = req.headers.get("Authorization") || ""; const callerClient = createClient( Deno.env.get("SUPABASE_URL"), Deno.env.get("SUPABASE_ANON_KEY"), { global: { headers: { Authorization: authHeader } } } ); const { data: { user }, error: authError } = await callerClient.auth.getUser(); if (authError || !user) { return new Response(JSON.stringify({ error: "No autenticado." }), { status: 401, headers: { ...cors, "Content-Type": "application/json" } }); } const { data: perfil } = await callerClient.from("profiles").select("role").eq("id", user.id).single(); if (!perfil || perfil.role !== "admin") { return new Response(JSON.stringify({ error: "Solo un administrador puede actualizar la TIIE manualmente." }), { status: 403, headers: { ...cors, "Content-Type": "application/json" } }); } const banxicoToken = Deno.env.get("BANXICO_TOKEN"); if (!banxicoToken) { return new Response(JSON.stringify({ error: "Falta configurar el secreto BANXICO_TOKEN en Supabase." }), { status: 500, headers: { ...cors, "Content-Type": "application/json" } }); } // Consulta el dato más reciente publicado de la TIIE 28 días. const url = `https://www.banxico.org.mx/SieAPIRest/service/v1/series/${SERIE_TIIE_28}/datos/oportuno`; const resp = await fetch(url, { headers: { "Bmx-Token": banxicoToken, "Accept": "application/json" } }); if (!resp.ok) { const txt = await resp.text(); return new Response(JSON.stringify({ error: `Banxico respondió ${resp.status}: ${txt}` }), { status: 502, headers: { ...cors, "Content-Type": "application/json" } }); } const json = await resp.json(); const serie = json?.bmx?.series?.[0]; const dato = serie?.datos?.[0]; if (!dato) { return new Response(JSON.stringify({ error: "Banxico no devolvió datos para la serie." }), { status: 502, headers: { ...cors, "Content-Type": "application/json" } }); } const tiie = parseFloat(dato.dato.replace(",", ".")); const fecha = dato.fecha; // formato DD/MM/AAAA // Guarda el valor en tasa_config (id=1) usando la service_role (solo disponible aquí, en el servidor). const supabase = createClient( Deno.env.get("SUPABASE_URL"), Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") ); const { error } = await supabase .from("tasa_config") .update({ tiie, tiie_fecha: fecha, tiie_auto: true, updated_at: new Date().toISOString() }) .eq("id", 1); if (error) { return new Response(JSON.stringify({ error: `Error guardando en Supabase: ${error.message}` }), { status: 500, headers: { ...cors, "Content-Type": "application/json" } }); } return new Response(JSON.stringify({ ok: true, tiie, fecha }), { headers: { ...cors, "Content-Type": "application/json" } }); } catch (e) { return new Response(JSON.stringify({ error: String(e) }), { status: 500, headers: { ...cors, "Content-Type": "application/json" } }); } });
